def feature_extract(img):
    img = Image.fromarray(cm.gray(img, bytes=True))
    img = img.resize((224, 224))
    img = img.convert('RGB')
    x = image.img_to_array(img)
    x = np.expand_dims(x, axis=0)
    x = preprocess_input(x)
    feature = model_fe.predict(x)[0]
    return feature
