Pre-Processing¶

Pre process this data. This uses ImageDataGenerator and data augmentation features to load the dataset. Augmentation was applied to increase size and variety of our dataset since it was relatively small.

In [238]:
from keras.preprocessing.image import ImageDataGenerator

# Your code here ...
train_dir = 'soa/train_folder'
validation_dir = 'soa/test_folder'

# All images will be rescaled by 1./255
train_datagen = ImageDataGenerator(rescale=1./255, rotation_range=20, zoom_range=0.15, width_shift_range=0.2, height_shift_range=0.2, shear_range=0.15, horizontal_flip=True)
test_datagen = ImageDataGenerator(rescale=1./255, rotation_range=20, zoom_range=0.15, width_shift_range=0.2, height_shift_range=0.2, shear_range=0.15, horizontal_flip=True)

train_generator = train_datagen.flow_from_directory(
        # This is the target directory
        train_dir,
        # All images will be resized to 150x150
        target_size=(32, 32),
        batch_size=20,
        # Since we use binary_crossentropy loss, we need binary labels
        class_mode='categorical')

validation_generator = test_datagen.flow_from_directory(
        validation_dir,
        target_size=(32, 32),
        batch_size=20,
        class_mode='categorical')

from keras import layers
from keras import models

# sequential
model = models.Sequential()

# layer 1 - convo2D specify input
model.add(layers.Conv2D(32, (3, 3), activation='relu',
                        input_shape=(32, 32, 3)))
# 32 is image size from above, (3, 3) is deimensions of kernel

# layer 2 - convo2D no input shape
model.add(layers.Conv2D(32, (3, 3), activation='relu'))

# layer 3 - maxpooling2D
model.add(layers.MaxPooling2D((2, 2)))

# layer 4 - dropout
model.add(layers.Dropout(rate = 0.5))

# layer 5 - flatten
model.add(layers.Flatten())

# layer 6 - dense
model.add(layers.Dense(128, activation='relu'))

# layer 7 - dropout
model.add(layers.Dropout(rate = 0.5))

# layer 8 - dense
model.add(layers.Dense(9, activation='softmax'))

def plot_image(i, predictions_array, true_labels, imgs):
    predictions_array, true_label, img = predictions_array, true_labels[i], imgs[i]
    
    plt.grid(False)
    plt.xticks([])
    plt.yticks([])

    plt.imshow(img, cmap=plt.cm.binary)
 
    predicted_label = np.argmax(predictions_array)
    if predicted_label == true_label:
        color = 'green'
    else:
        color = 'red'

    plt.xlabel("{} {:2.0f}% ({})".format(class_names[predicted_label],
                                100*np.max(predictions_array),
                                class_names[true_label]),
                                color=color)

def plot_value_array(i, predictions_array, true_labels):
    predictions_array, true_label = predictions_array, true_labels[i]
    plt.grid(False)
    plt.xticks(range(9))
    plt.yticks([])
    thisplot = plt.bar(range(9), predictions_array, color="#777777")
    plt.ylim([0, 1])
    predicted_label = np.argmax(predictions_array)

    thisplot[predicted_label].set_color('red')
    thisplot[true_label].set_color('green')
